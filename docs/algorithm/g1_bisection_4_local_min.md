func FindLocalMinIndex(arr []int) int {
   if arr == nil || len(arr) < 2 {
      return -1
   }
   if arr[0] < arr[1] {
      return 0
   }
   if arr[len(arr)-2] > arr[len(arr)-1] {
      return len(arr) - 1
   }
   left, right, mid := 0, len(arr)-1, 0
   for left < right {
      mid = left + ((right - left) >> 1)
      if arr[mid-1] > arr[mid] && arr[mid+1] > arr[mid] {
         return mid
      } else if  arr[mid-1] < arr[mid] {
         right = mid - 1
      } else {
         left = mid + 1
      }
   }
   return left
}
